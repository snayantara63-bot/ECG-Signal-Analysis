import numpy as np
import matplotlib.pyplot as plt

file_path = r"C:\Users\LENOVO\OneDrive\Documents\ecg_signal.txt"

time = []
ecg = []

with open(file_path, "r") as f:
    for line in f:
        parts = line.strip().split()
        if len(parts) == 2:
            try:
                t = float(parts[0])
                v = float(parts[1])
                time.append(t)
                ecg.append(v)
            except:
                pass

time = np.array(time)
ecg = np.array(ecg)

plt.figure(figsize=(10,4))
plt.plot(time, ecg)
plt.xlabel("Time (min)")
plt.ylabel("ECG (mV)")
plt.title("ECG Signal")
plt.grid(True)
plt.show()

import numpy as np
import matplotlib.pyplot as plt

fs = 360  # sampling frequency

N = len(ecg)
ecg_fft = np.fft.fft(ecg)
freq = np.fft.fftfreq(N, d=1/fs)

positive_freqs = freq[:N//2]
fft_magnitude = np.abs(ecg_fft[:N//2]) * 2 / N

plt.figure(figsize=(10,4))
plt.plot(positive_freqs, fft_magnitude)
plt.xlabel("Frequency (Hz)")
plt.ylabel("Magnitude")
plt.title("Single-Sided Frequency Spectrum of ECG")
plt.xlim(0, 50)
plt.grid(True)
plt.show()
